def self.interpret(command)
  responses = []

  if command.match(/^(?:show|me|whats|what|is|my|\s)*OS(?:\s|name|do i have|is used|.)*$/i)
    os = platform_name

    responses << {
      :command => "echo '#{os}'",
      :explanation => "Show what OS is used."
    }
  end

  responses
end

def self.platform_name
  os = ""

  case RUBY_PLATFORM
    when /cygwin|mswin|mingw|bccwin|wince|emx/
      os = "Windows"
    when /darwin/
      os = "OS X"
    when /freebsd/
      os = "FreeBSD"
    when /openbsd/
      os = "OpenBSD"
    when /netbsd/
      os = "NetBSD"
    when /linux/
      os = "Linux"
  end

  os
end
